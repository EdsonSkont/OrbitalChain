\section{Satellite-Adapted SBFT Consensus (SA-SBFT)}

\subsection{Motivation and Adaptations}

The original SBFT consensus was designed for enterprise dispute resolution with stable network topology and consistent connectivity. Adapting it for LEO satellite networks requires addressing several unique challenges:

\begin{enumerate}
    \item \textbf{Dynamic Topology:} Satellite positions change continuously due to orbital mechanics
    \item \textbf{Intermittent Connectivity:} Satellites have limited visibility windows
    \item \textbf{Variable Latency:} Link quality varies with distance and atmospheric conditions
    \item \textbf{Energy Constraints:} Satellites have limited power budgets
    \item \textbf{Geographic Sharding:} Natural network partitioning by coverage area
\end{enumerate}

\subsection{Key Improvements for Satellite Networks}

\subsubsection{1. Orbital-Aware Primary Selection}

Instead of selecting the primary node solely based on reputation, we incorporate orbital visibility predictions:

\begin{definition}[Orbital Reliability Score]
The orbital reliability score for satellite $S_i$ at time $t$ is:
\begin{equation}
\mathcal{R}^{orb}_i(t) = R_i \cdot \underbrace{\frac{T^{vis}_i(t)}{T_{epoch}}}_{\text{visibility factor}} \cdot \underbrace{\exp\left(-\frac{d_i(t)}{\bar{d}}\right)}_{\text{distance factor}} \cdot \underbrace{\frac{E_i(t)}{E_{max}}}_{\text{energy factor}}
\end{equation}
where:
\begin{itemize}
    \item $R_i$ is the reputation value (from original SBFT)
    \item $T^{vis}_i(t)$ is predicted visibility duration within current epoch
    \item $d_i(t)$ is average distance to other satellites in shard
    \item $E_i(t)$ is remaining energy budget
\end{itemize}
\end{definition}

\subsubsection{2. Predictive View Change Protocol}

Original SBFT triggers view changes reactively when the primary fails. For satellites, we implement predictive view changes based on orbital predictions:

\begin{algorithm}[H] \small
\caption{Predictive View Change for Satellite Networks}
\label{alg:predictive_view_change}
\begin{algorithmic}[1]
\State $T_{predict} \leftarrow$ 30 seconds \Comment{Prediction horizon}
\State $\theta_{handover} \leftarrow 5°$ \Comment{Elevation threshold for handover}

\For{each epoch $t$}
    \State $\vec{r}_{APN}(t + T_{predict}) \leftarrow$ PropagateOrbit($APN$, $t + T_{predict}$)
    \State $\theta_{APN} \leftarrow$ ComputeElevation($\vec{r}_{APN}$, shard\_center)
    
    \If{$\theta_{APN} < \theta_{handover}$}
        \State \textbf{/* Initiate proactive view change */}
        \State $candidates \leftarrow \{S_i \in \mathbb{A} : \theta_i(t + T_{predict}) > \theta_{min}\}$
        \State $APN_{new} \leftarrow \arg\max_{S_i \in candidates} \mathcal{R}^{orb}_i(t)$
        \State InitiateHandover($APN$, $APN_{new}$)
    \EndIf
\EndFor
\end{algorithmic}
\end{algorithm}

\subsubsection{3. Energy-Aware Role Assignment}

Satellites have limited energy budgets. We extend the Active/Semi-Active classification with energy awareness:

\begin{equation}
\text{Role}(S_i) = 
\begin{cases}
\text{Active} & \text{if } R_i \geq \lambda \text{ AND } E_i \geq E_{active} \\
\text{Semi-Active} & \text{if } R_i \geq 0.1 \text{ AND } E_i \geq E_{semi} \\
\text{Dormant} & \text{if } E_i < E_{semi}
\end{cases}
\end{equation}

Dormant satellites participate in consensus verification but do not actively broadcast, conserving energy for critical operations.

\subsubsection{4. Inter-Satellite Link (ISL) Aware Communication}

Unlike terrestrial networks, satellite communication uses ISLs with varying quality. We optimize message routing:

\begin{equation}
\text{Cost}_{ij}(t) = \frac{1}{R^{ISL}_{ij}(t)} + \alpha \cdot \text{Delay}_{ij}(t) + \beta \cdot \text{BER}_{ij}(t)
\end{equation}

where $R^{ISL}_{ij}$ is the ISL data rate, $\text{Delay}_{ij}$ is propagation delay, and $\text{BER}_{ij}$ is bit error rate.

\subsection{Improved SA-SBFT Algorithm}

\begin{algorithm}[H] \small
\caption{Satellite-Adapted SBFT (SA-SBFT) Consensus}
\label{alg:sa_sbft}
\textbf{Input:} Transaction batch $\mathcal{TX}$, Satellite shard $\mathcal{S}$, Current time $t$ \\
\textbf{Output:} Consensus decision, Block commitment
\begin{algorithmic}[1]

\Statex \textbf{/* Phase 0: Orbital-Aware Node Classification */}
\For{each satellite $S_i \in \mathcal{S}$}
    \State $\mathcal{R}^{orb}_i \leftarrow$ ComputeOrbitalReliability($S_i$, $t$)
    \State $\theta_i \leftarrow$ GetElevationAngle($S_i$, shard\_center)
    \State $E_i \leftarrow$ GetEnergyLevel($S_i$)
    
    \If{$\mathcal{R}^{orb}_i \geq \lambda_{orb}$ \textbf{AND} $\theta_i > \theta_{min}$ \textbf{AND} $E_i \geq E_{active}$}
        \State $\mathbb{A} \leftarrow \mathbb{A} \cup \{S_i\}$ \Comment{Active Satellite}
    \ElsIf{$\mathcal{R}^{orb}_i \geq 0.1$ \textbf{AND} $E_i \geq E_{semi}$}
        \State $\mathbb{S} \leftarrow \mathbb{S} \cup \{S_i\}$ \Comment{Semi-Active Satellite}
    \Else
        \State $\mathbb{D} \leftarrow \mathbb{D} \cup \{S_i\}$ \Comment{Dormant - verify only}
    \EndIf
\EndFor

\State $N_a \leftarrow |\mathbb{A}|$, $N_s \leftarrow |\mathbb{S}|$, $N \leftarrow N_a + N_s$
\State \textbf{Require:} $N \geq 3f + 1$ where $f = \lfloor(N-1)/3\rfloor$

\Statex \textbf{/* Phase 1: Primary Selection with Orbital Awareness */}
\State $APN \leftarrow \arg\max_{S_i \in \mathbb{A}} \mathcal{R}^{orb}_i$
\State $backup\_list \leftarrow$ Sort($\mathbb{A} \setminus \{APN\}$, key=$\mathcal{R}^{orb}$, descending)

\Statex \textbf{/* Phase 2: Request Phase */}
\State $APN$ computes optimal ISL routing tree $\mathcal{T}^*$
\State $msg_{req} \leftarrow \langle \text{REQUEST}, \mathcal{TX}, H(\mathcal{TX}), t, \text{MAC}_{APN} \rangle$
\State $APN$ broadcasts $msg_{req}$ via $\mathcal{T}^*$

\Statex \textbf{/* Phase 3: Pre-Prepare with Parallel Verification */}
\For{each $S_i \in \mathbb{A} \cup \mathbb{S}$ \textbf{in parallel}}
    \State Verify $msg_{req}$ authenticity
    \State $decision_i \leftarrow$ ValidateTransactions($\mathcal{TX}$)
    \State $msg_{pp} \leftarrow \langle \text{PRE-PREPARE}, decision_i, H(\mathcal{TX}), \text{MAC}_i \rangle$
    
    \If{$S_i \in \mathbb{A}$}
        \State Broadcast $msg_{pp}$ to all $\mathbb{A}$ \Comment{Active-to-Active}
    \Else
        \State Send $msg_{pp}$ to nearest $k$ Active satellites \Comment{Semi-to-Active, $k=\lceil N_a/2 \rceil$}
    \EndIf
\EndFor

\Statex \textbf{/* Phase 4: Prepare with Early Termination */}
\State $votes \leftarrow$ CollectVotes(timeout=$T_{ISL} \cdot 2$)
\State $majority \leftarrow$ GetMajorityDecision($votes$)

\If{$|votes_{majority}| \geq 2f + 1$}
    \State \textbf{/* Early consensus - skip additional rounds */}
    \For{each $S_i \in \mathbb{A}$}
        \If{$decision_i \neq majority$}
            \State $decision_i \leftarrow majority$
            \State FlagForReputationDecrease($S_i$)
        \EndIf
        \State $msg_p \leftarrow \langle \text{PREPARE}, majority, H(\mathcal{TX}), \text{MAC}_i \rangle$
        \State Broadcast $msg_p$ to $\mathbb{A}$
    \EndFor
\Else
    \State \Return ConsensusFailure
\EndIf

\Statex \textbf{/* Phase 5: Commit with Efficient Aggregation */}
\State $APN$ waits for $2f$ PREPARE confirmations
\State $\sigma_{agg} \leftarrow$ AggregateSignatures($\{MAC_i : S_i \in \mathbb{A}\}$) \Comment{BLS aggregation}
\State $msg_c \leftarrow \langle \text{COMMIT}, majority, H(\mathcal{TX}), \sigma_{agg} \rangle$
\State $APN$ broadcasts $msg_c$ via $\mathcal{T}^*$

\Statex \textbf{/* Phase 6: Reply with Sampling Verification */}
\For{each $S_i \in \mathbb{A} \cup \mathbb{S}$}
    \State Verify $\sigma_{agg}$
    \State $msg_r \leftarrow \langle \text{REPLY}, majority, \text{MAC}_i \rangle$
    \State Send $msg_r$ to $APN$
\EndFor

\State $APN$ samples $\lceil N/3 \rceil$ replies for consistency check
\If{Consistent}
    \State CommitBlock($\mathcal{TX}$, $majority$)
\EndIf

\Statex \textbf{/* Phase 7: Reputation and Energy Update */}
\For{each satellite $S_i$}
    \If{$decision_i = majority$}
        \State $R_i(t+1) \leftarrow \frac{1}{2}[R_i(t)(1-a) + a] + \frac{1}{2}$
    \Else
        \State $R_i(t+1) \leftarrow R_i(t)(1-b)$ \Comment{$b > a$}
    \EndIf
    \State UpdateEnergyConsumption($S_i$, role)
\EndFor

\State \Return $majority$, Block
\end{algorithmic}
\end{algorithm}


\subsection{Complexity Analysis}

\subsubsection{Message Complexity}

\begin{theorem}[SA-SBFT Message Complexity]
The SA-SBFT protocol achieves message complexity $O(N \cdot N_a)$, identical to the original SBFT, while adapting to satellite network dynamics.
\end{theorem}

\begin{proof}
Message count per phase:
\begin{itemize}
    \item Request: 1 (broadcast via ISL tree)
    \item Pre-prepare: $N_a(N_a-1) + N_s \cdot k$ where $k = \lceil N_a/2 \rceil$
    \item Prepare: $N_a(N_a-1)$
    \item Commit: $N-1$ (via aggregated signature)
    \item Reply: $N$
\end{itemize}
Total: $O(N_a^2 + N_s \cdot N_a) = O(N \cdot N_a)$
\end{proof}

\subsubsection{Communication Latency}

\begin{theorem}[Consensus Latency Bound]
The expected consensus latency is bounded by:
\begin{equation}
T_{consensus} \leq 4 \cdot \max_{i,j \in \mathcal{S}} \text{RTT}_{ij} + T_{verify} + T_{aggregate}
\end{equation}
where $\text{RTT}_{ij}$ is the round-trip time between satellites $i$ and $j$.
\end{theorem}

For LEO satellites at 550 km altitude with ISL, typical $\text{RTT} \approx 10-50$ ms, yielding consensus latency of 100-300 ms.

\subsection{Optimizations for Satellite Environment}

\subsubsection{1. BLS Signature Aggregation}

To reduce communication overhead, we use Boneh-Lynn-Shacham (BLS) signatures that can be aggregated:

\begin{equation}
\sigma_{agg} = \prod_{i \in \mathbb{A}} \sigma_i
\end{equation}

Verification: $e(\sigma_{agg}, g) \stackrel{?}{=} \prod_{i \in \mathbb{A}} e(H(m), pk_i)$

This reduces Commit phase message size from $O(N)$ signatures to 1 aggregated signature.

\subsubsection{2. Pipelined Consensus}

For high-throughput scenarios, we pipeline consensus rounds:

\begin{equation}
\text{Throughput}_{pipeline} = \frac{B}{T_{phase}} \text{ vs } \text{Throughput}_{sequential} = \frac{B}{T_{consensus}}
\end{equation}

where $B$ is block size and $T_{phase}$ is single phase duration.

\subsubsection{3. Checkpoint-Based Recovery}

Satellites may lose connectivity temporarily. We implement lightweight checkpointing:

\begin{algorithm}[H] \small
\caption{Checkpoint and Recovery Protocol}
\begin{algorithmic}[1]
\State \textbf{Every} $K$ consensus rounds:
    \State $checkpoint \leftarrow \langle block\_height, state\_root, \sigma_{2f+1} \rangle$
    \State Store $checkpoint$ with $2f+1$ signatures
\State
\State \textbf{On satellite reconnection:}
    \State Request latest $checkpoint$ from $f+1$ peers
    \State Fast-forward state to $checkpoint$
    \State Resume normal consensus participation
\end{algorithmic}
\end{algorithm}


\subsection{Security Analysis}

\subsubsection{Byzantine Fault Tolerance}

\begin{theorem}[SA-SBFT Safety]
SA-SBFT maintains safety (agreement consistency) if $f < N/3$ satellites are Byzantine.
\end{theorem}

\begin{proof}
The proof follows from standard PBFT analysis, extended for satellite dynamics:

1. \textbf{Quorum Intersection:} Any two quorums of size $2f+1$ intersect in at least $f+1$ honest satellites.

2. \textbf{View Change Safety:} Predictive view changes preserve safety because:
   - New primary is selected before old primary loses visibility
   - State transfer occurs during overlap period
   - $2f+1$ satellites must acknowledge new view

3. \textbf{Orbital Partition Recovery:} If network partitions due to orbital mechanics:
   - No partition can have $2f+1$ satellites (by BFT assumption)
   - Consensus halts (liveness affected) but safety preserved
   - Recovery occurs when visibility restored
\end{proof}

\subsubsection{Liveness Guarantee}

\begin{theorem}[SA-SBFT Liveness]
Under partial synchrony with bounded message delay $\Delta$ and $f < N/3$ Byzantine faults, SA-SBFT eventually reaches consensus.
\end{theorem}

The predictive view change mechanism ensures liveness by:
\begin{itemize}
    \item Preventing primary unavailability due to orbital mechanics
    \item Maintaining $\geq 2f+1$ visible satellites per shard
    \item Adapting timeout parameters to ISL latency
\end{itemize}


\subsection{Integration with Truth Discovery}

For OrbitalChain's privacy-preserving truth discovery, SA-SBFT consensus is used to:

\begin{enumerate}
    \item \textbf{Intra-Shard Consensus:} Agree on aggregated truth values within each satellite shard
    \item \textbf{Inter-Shard Consensus:} Coordinate between shards for global consistency
    \item \textbf{MPC Output Verification:} Validate that garbled circuit outputs are consistent
\end{enumerate}

\begin{algorithm}[H] \small
\caption{Truth Discovery with SA-SBFT Integration}
\begin{algorithmic}[1]

\State $[\Phi_l], [w_k] \leftarrow$ SecureTruthDiscovery($\{[x_k]\}$) \Comment{Algorithm 2}

\State $\mathcal{TX}_{truth} \leftarrow \langle \Phi_l, \{w_k\}, H(shares), epoch \rangle$
\State $(decision, block) \leftarrow$ SA\_SBFT($\mathcal{TX}_{truth}$)

\If{$decision = $ COMMIT}
    \State Record $\Phi_l$ to blockchain
    \State Trigger D-Stream clustering with $\Phi_l$
\EndIf
\end{algorithmic}
\end{algorithm}

def build_isl_routing_tree(apn, satellites, t):
    """Build minimum-cost spanning tree for ISL routing."""
    import networkx as nx
    
    G = nx.Graph()
    for si in satellites:
        for sj in satellites:
            if si != sj and has_isl_link(si, sj, t):
                cost = compute_isl_cost(si, sj, t)
                G.add_edge(si, sj, weight=cost)
    
    # Minimum spanning tree rooted at APN
    T = nx.minimum_spanning_tree(G)
    return nx.bfs_tree(T, apn)
\end{lstlisting}
